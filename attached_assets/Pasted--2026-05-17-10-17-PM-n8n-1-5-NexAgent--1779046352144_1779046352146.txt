      """Start background thread to clean up idle conversation contexts."""
      import threading, time
      def _cleanup_loop():
          while self._running:
              time.sleep(1800)
              now = time.time()
              idle_cutoff = 7200
              to_remove = [
                  uid for uid, last in self.last_activity.items()
                  if now - last > idle_cutoff
              ]
              for uid in to_remove:
                  self.conversation_context.pop(uid, None)
                  self.last_activity.pop(uid, None)
              if to_remove:

                      f"Context cleanup: removed {len(to_remove)} idle users.")
