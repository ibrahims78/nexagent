║       PC Commander - دليل إعداد Pre-Login Agent              ║
║         الإشعار والتحكم عند شاشة تسجيل الدخول              ║
╚══════════════════════════════════════════════════════════════╝

ما الذي يفعله هذا السكريبت؟
━━━━━━━━━━━━━━━━━━━━━━━━━━━
عند كل إقلاع للحاسب، يعمل هذا السكريبت قبل تسجيل الدخول
ويرسل إشعاراً على تيليغرام يخبرك أن الحاسب جاهز وينتظرك.
يمكنك عندها إرسال أمر تسجيل الدخول من أي مكان في العالم.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الطريقة 1: Autologon (تسجيل دخول تلقائي بالكامل)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الأفضل لـ: من يريد الحاسب يدخل ويندوز تلقائياً بدون أي تدخل

الإعداد:
1. افتح PC Commander
2. اذهب لتبويب "تسجيل الدخول"
3. أدخل اسم المستخدم وكلمة المرور
4. اضغط "تفعيل Autologon"
5. سيتم تحميل أداة Autologon من Microsoft تلقائياً وضبطها

ما يحدث عند الإقلاع:
  ← ويندوز يبدأ
  ← Autologon يسجّل الدخول تلقائياً
  ← البوت الرئيسي يعمل
  ← إشعار "✅ الحاسب جاهز"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الطريقة 2: Pre-Login Agent (تسجيل دخول عند الطلب)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الأفضل لـ: من يريد التحكم بوقت تسجيل الدخول يدوياً

الإعداد:

الخطوة 1 - ضبط الإعدادات:
  افتح ملف pre_login_config.json وعدّله:
  {
    "bot_token": "نفس Token بوت PC Commander",
    "allowed_users": [123456789],  ← User ID الخاص بك
    "notify_on_boot": true,
    "auto_login_enabled": false,
    "command_timeout": 300         ← انتظار 5 دقائق
  }

الخطوة 2 - التثبيت:
  1. انقر بالزر الأيمن على install_pre_login.bat
  2. اختر "تشغيل كمسؤول"
  3. انتظر اكتمال التثبيت

الخطوة 3 - الاختبار:
  1. أعد تشغيل الحاسب
  2. ستصلك رسالة تيليغرام تقول "الحاسب أُقلع وينتظر"
  3. اضغط "سجّل الدخول الآن"
  4. أرسل كلمة مرورك (ستُحذف تلقائياً)
  5. سيدخل ويندوز

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الجمع بين الطريقتين (الأفضل)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ← Autologon مفعّل: الحاسب يدخل تلقائياً دون انتظار
  ← Pre-Login Agent يعمل: يرسل لك إشعاراً بنجاح الإقلاع
  ← البوت الرئيسي يعمل: جاهز للتحكم

هذا الإعداد يعطيك:
  ✅ تسجيل دخول تلقائي بدون تدخل
  ✅ إشعار فوري عند جاهزية الحاسب
  ✅ تحكم كامل فوراً بعد الإقلاع

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 إلغاء التثبيت
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Pre-Login Agent:
  شغّل uninstall_pre_login.bat كمسؤول

Autologon:
  في PC Commander → تبويب "تسجيل الدخول" → "إلغاء Autologon"
  أو من سطر الأوامر: Autologon.exe /accepteula (بدون كلمة مرور)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ملاحظات أمنية
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚠️ Autologon يخزن كلمة المرور بشكل مشفر في الـ Registry
⚠️ Pre-Login Agent يحذف رسالة كلمة المرور فور استخدامها
⚠️ لا يُنصح باستخدام Autologon إذا كان الحاسب في بيئة عامة
✅ مناسب للاستخدام المنزلي بأمان
