  ما هو SshRemote؟
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  SshRemote هو الطبقة التحتية لمشروع NexAgent.
  يتيح الاتصال بجهاز Windows عن بُعد عبر SSH من أي مكان،
  حتى لو كان الجهاز خلف Router بدون IP ثابت.

  كيف يعمل:
  1. يفتح bore نفقاً عاماً على bore.pub
  2. يرسل رقم المنفذ (Port) لتيليغرام تلقائياً
  3. تتصل من أي مكان بأمر SSH بسيط

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  متطلبات التثبيت
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  - نظام تشغيل: Windows 10 أو Windows 11
  - صلاحيات: مدير النظام (Administrator)
  - اتصال إنترنت فعّال
  - حساب تيليغرام + بوت (الإنشاء مشروح في الخطوة 0)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 0: إنشاء بوت تيليغرام (إذا ما عندك واحد)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  [ إنشاء البوت والحصول على التوكن ]
  1. افتح تيليغرام وابحث عن: @BotFather
  2. أرسل: /newbot
  3. أدخل اسماً للبوت (مثال: My NexAgent Bot)
  4. أدخل username ينتهي بـ "bot" (مثال: mynexagent_bot)
  5. احفظ التوكن الذي يرسله BotFather

  [ الحصول على Chat ID ]
  1. افتح تيليغرام وابحث عن: @userinfobot
  2. أرسل: /start
  3. احفظ الرقم الذي يرسله — هذا هو الـ Chat ID

  ⚠️ أرسل /start للبوت الجديد من حسابك قبل التثبيت.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 1: الإعداد المسبق
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. افتح ملف: sshremote_config.ini  (بـ Notepad)
  2. عبّئ الحقلين:

     bot_token = ضع توكن البوت هنا
     chat_id   = ضع الـ Chat ID هنا

  3. احفظ الملف.
  4. (اختياري) غيّر install_path إذا أردت مسار تثبيت مختلف
     الافتراضي: C:\SshRemote_V2

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 2: التثبيت
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. انقر بزر الأيمن على: setup_v2.bat
  2. اختر: "تشغيل كمسؤول" (Run as administrator)
  3. انتظر اكتمال الخطوات السبع — كل خطوة تظهر "OK"
  4. عند ظهور "Setup Complete!" — التثبيت نجح
  5. تحقق من تيليغرام — ستصل رسالة خلال 30 ثانية

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الخطوة 3: الاتصال عن بُعد
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  بعد استلام رسالة تيليغرام:

  [ Windows — PowerShell أو CMD ]
    ssh -i .\sshremote_key USERNAME@bore.pub -p PORT

  [ Mac / Linux — Terminal ]
    chmod 600 sshremote_key
    ssh -i ./sshremote_key USERNAME@bore.pub -p PORT

  مثال:
    ssh -i sshremote_key ibrahim@bore.pub -p 54321

  ملفات المشروع
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  sshremote_config.ini     ← إعدادات التيليغرام والتثبيت  ← عبّئه أولاً
  setup_v2.bat             ← التثبيت الكامل (مرة واحدة كمسؤول)
  start_tunnel_v2.bat      ← تشغيل النفق يدوياً
  stop_tunnel_v2.bat       ← إيقاف النفق يدوياً
  show_port_v2.bat         ← عرض رقم المنفذ الحالي
  uninstall_v2.bat         ← إزالة كاملة نظيفة
  sshremote_tunnel_v2.ps1  ← سكريبت النفق الرئيسي (تلقائي)
  sshremote_key.pub        ← المفتاح العام (آمن للمشاركة)
  bore.exe                 ← أداة النفق
  sshremote_key            ← ⚠️ المفتاح الخاص — لا تشاركه أبداً

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  إدارة النفق
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  تشغيل  : زر أيمن على start_tunnel_v2.bat  → Run as administrator
  إيقاف  : زر أيمن على stop_tunnel_v2.bat   → Run as administrator
  المنفذ : شغّل show_port_v2.bat (بدون admin)
  إزالة  : زر أيمن على uninstall_v2.bat     → Run as administrator

  ملاحظة: النفق يبدأ تلقائياً عند كل تسجيل دخول لويندوز.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  حل المشاكل الشائعة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  لا تصل رسالة تيليغرام
  → تحقق من bot_token وchat_id في sshremote_config.ini
  → أرسل /start للبوت من حسابك
  → افحص: C:\SshRemote_V2\sshremote_tunnel.log

  "Access Denied" أثناء التثبيت
  → زر أيمن على setup_v2.bat → "تشغيل كمسؤول"

  "Connection refused" عند SSH
  → شغّل show_port_v2.bat للتحقق من المنفذ
  → تأكد أن رسالة تيليغرام وصلت أولاً

  "Permission denied (publickey)"
  → تأكد من استخدام: ssh -i sshremote_key ...
  → على Mac/Linux: chmod 600 sshremote_key

  bore يتوقف
  → يعيد المحاولة تلقائياً حتى 10 مرات
  → إذا فشل كل المحاولات ستصل رسالة تيليغرام
  → شغّل stop ثم start يدوياً

  ملاحظات أمنية
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ⚠️ ملف sshremote_key يمنح وصولاً كاملاً للجهاز — لا تشاركه.
  ⚠️ عند الشك بالاختراق: شغّل uninstall_v2.bat فوراً.
  ⚠️ أوقف النفق عند عدم الحاجة بـ stop_tunnel_v2.bat.
  ⚠️ لا تثبّت على جهاز لا تملكه بدون إذن صاحبه.
