هذا الدليل لإعداد هاتف أندرويد قديم (أو هاتف شخص في المنزل)
حتى يتمكن من تشغيل الحاسب عن بُعد عبر تيليغرام.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 1: تفعيل Wake-on-LAN على الحاسب
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أولاً - من BIOS/UEFI:
1. أعد تشغيل الحاسب واضغط F2 أو DEL أو F10 (حسب اللوحة الأم)
2. ابحث عن: "Wake on LAN" أو "Power On By PCI-E" أو "WOL"
3. فعّله (Enabled)
4. احفظ وأعد التشغيل

ثانياً - من ويندوز:
1. افتح Device Manager (مدير الأجهزة)
2. افتح Network Adapters
3. انقر بالزر الأيمن على بطاقة الشبكة → Properties
4. اذهب لتبويب Power Management
5. فعّل: "Allow this device to wake the computer"
6. فعّل: "Only allow a magic packet to wake the computer"
7. اضغط OK

ثالثاً - الحصول على MAC Address للحاسب:
1. افتح CMD واكتب: ipconfig /all
2. ابحث عن "Physical Address" تحت بطاقة الشبكة
   مثال: Physical Address: A1-B2-C3-D4-E5-F6
3. احفظ هذا الرقم - ستحتاجه لاحقاً

رابعاً - الحصول على IP الحاسب:
1. في نفس نتيجة ipconfig ابحث عن "IPv4 Address"
   مثال: 192.168.1.100
2. يُفضّل جعل IP ثابتاً من إعدادات الراوتر (IP Reservation)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 2: تثبيت Termux على الهاتف
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚠️ هام: لا تثبت Termux من Google Play (نسخة قديمة)
بدلاً من ذلك:

الطريقة أ - من F-Droid (الأفضل):
1. حمّل F-Droid من: f-droid.org
2. افتح F-Droid وابحث عن "Termux"
3. ثبّته

الطريقة ب - APK مباشر:
1. اذهب لـ: github.com/termux/termux-app/releases
2. حمّل آخر إصدار APK
3. ثبّته (قد تحتاج تفعيل "تثبيت من مصادر غير معروفة")

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 3: إعداد Termux
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

افتح Termux والصق هذه الأوامر واحداً تلو الآخر:

1. تحديث الحزم:
   pkg update -y && pkg upgrade -y

2. تثبيت Python:
   pkg install python -y

3. تثبيت المكتبة المطلوبة:
   pip install python-telegram-bot==20.7

4. إنشاء مجلد للبرنامج:
   mkdir -p ~/wol_agent
   cd ~/wol_agent

5. نسخ ملفات البرنامج:
   (انقل ملفات wol_agent.py و wol_config.json إلى الهاتف)
   
   يمكنك نسخها عبر:
   - USB من الحاسب
   - تطبيق Files أو مدير الملفات
   - أرسلها لنفسك عبر تيليغرام وحمّلها

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 4: إنشاء بوت تيليغرام احتياطي
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚠️ يجب إنشاء بوت جديد منفصل (غير بوت PC Commander)

1. افتح تيليغرام وابحث عن @BotFather
2. أرسل: /newbot
3. أدخل اسم البوت (مثلاً: Home WoL Bot)
4. أدخل username (مثلاً: home_wol_bot)
5. احفظ الـ Token الذي يعطيك إياه

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 5: ضبط الإعدادات
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

افتح ملف wol_config.json وعدّل القيم:

{
  "bot_token": "ضع هنا Token البوت الاحتياطي",
  "allowed_users": [123456789],   ← User IDs المسموح لهم
  "pc_mac": "A1:B2:C3:D4:E5:F6", ← MAC Address الحاسب
  "pc_broadcast": "192.168.1.255",← آخر خانة في IP الشبكة = 255
  "pc_ip": "192.168.1.100",       ← IP الحاسب
  "agent_name": "وكيل المنزل"
}

مثال على Broadcast IP:
- إذا IP الحاسب: 192.168.1.100 → Broadcast: 192.168.1.255
- إذا IP الحاسب: 192.168.0.50  → Broadcast: 192.168.0.255
- إذا IP الحاسب: 10.0.0.5      → Broadcast: 10.0.0.255

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 6: تشغيل البرنامج
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

في Termux:
   cd ~/wol_agent
   python wol_agent.py

للتشغيل التلقائي في الخلفية دائماً:
1. ثبّت Termux:Boot من F-Droid
2. افتح Termux:Boot مرة واحدة
3. في Termux نفذ:
   mkdir -p ~/.termux/boot
   
4. أنشئ ملف التشغيل التلقائي:
   echo '#!/data/data/com.termux/files/usr/bin/sh
   cd ~/wol_agent
   python wol_agent.py &' > ~/.termux/boot/start_wol.sh
   chmod +x ~/.termux/boot/start_wol.sh

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 7: منع أندرويد من إيقاف البرنامج
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. اذهب لـ الإعدادات → البطارية
2. ابحث عن Termux
3. اختر "لا تُحسّن" أو "بدون قيود"
4. تأكد من إلغاء تفعيل "تحسين البطارية" لـ Termux

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 الخطوة 8: ربطه مع PC Commander الرئيسي
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

في إعدادات PC Commander على الحاسب:
1. افتح تبويب "إيقاظ الحاسب (WoL)"
2. أدخل MAC Address الحاسب
3. أدخل IP الحاسب
4. في "المستخدمون الاحتياطيون" أدخل User ID الشخص في المنزل
5. اضغط "حفظ"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 اختبار الإعداد
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. أغلق حاسبك تماماً
2. تأكد أن الهاتف متصل بنفس الواي فاي المنزلي
3. افتح بوت الهاتف الاحتياطي
4. اضغط "تشغيل الحاسب"
5. انتظر 30-60 ثانية
6. اضغط "هل الحاسب يعمل؟"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ملاحظة مهمة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚠️ Wake-on-LAN يعمل فقط عبر الشبكة المحلية (واي فاي المنزل)
⚠️ الهاتف يجب أن يكون متصلاً بنفس الواي فاي المنزلي
⚠️ إذا كنت خارج المنزل، ترسل الأمر عبر تيليغرام للهاتف
   الذي هو في المنزل، والهاتف يرسل الإشارة للحاسب
